Modules

Herbruikbare code blokken

Wat zijn Modules?

Modules zijn bestanden met Python code die je kunt hergebruiken in andere programma's. Ze bevatten functies, klassen en variabelen die je kunt importeren.

Voordelen van modules

  • Herbruikbaarheid: Code kan in meerdere programma's gebruikt worden
  • Organisatie: Grote programma's worden opgedeeld in kleinere delen
  • Onderhoudbaarheid: Makkelijker om code te wijzigen
  • Namespaces: Voorkomt naamconflicten
  • Distributie: Code kan eenvoudig gedeeld worden

Built-in modules

# math module voor wiskundige functies
import math

print(math.sqrt(16))    # 4.0
print(math.pi)          # 3.141592653589793
print(math.sin(math.pi/2))  # 1.0

# datetime module voor datums en tijden
import datetime

nu = datetime.datetime.now()
print(nu)               # 2024-01-15 14:30:45.123456
print(nu.year)          # 2024
print(nu.strftime("%d-%m-%Y"))  # 15-01-2024

# random module voor willekeurige getallen
import random

print(random.randint(1, 10))  # Willekeurig getal tussen 1 en 10
print(random.choice(["appel", "banaan", "kers"]))  # Willekeurige keuze

Importeren

Modules gebruiken

Modules Importeren

Verschillende manieren om te importeren

# Volledige module importeren
import math
resultaat = math.sqrt(25)

# Specifieke functies importeren
from math import sqrt, pi
resultaat = sqrt(25)

# Alias gebruiken
import math as m
resultaat = m.sqrt(25)

# Alles importeren (niet aanbevolen)
from math import *
resultaat = sqrt(25)

Voorbeelden met verschillende modules

# os module voor operating system functies
import os

print(os.getcwd())      # Huidige directory

try:

except:

Eenvoudige module maken

Maak een bestand mijn_module.py met de volgende inhoud:

# mijn_module.py
def groet(naam):
    """Een functie die groet"""
    return f"Hallo {naam}!"

    """Telt twee getallen op"""
    return a + b

PI = 3.14159

class Calculator:
    """Een eenvoudige rekenmachine klasse"""
    def multiply(self, a, b):
        return a * b

Module gebruiken

# In een ander bestand
import mijn_module

print(mijn_module.groet("Lucas"))  # Hallo Lucas!
print(mijn_module.tel_op(5, 3))    # 8
print(mijn_module.PI)              # 3.14159

calc = mijn_module.Calculator()
print(calc.multiply(4, 5))         # 20

Games en GUI Libraries

Pygame voor games

import pygame
import sys

# Pygame initialiseren
pygame.init()

# Scherm instellen
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("Mijn Eerste Game")

# Kleuren
WHITE = (255, 255, 255)
BLUE = (0, 0, 255)

# Speler positie
player_x = 400
player_y = 300

# Game loop
running = True
while running:
    # Events afhandelen
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # Toetsen afhandelen
    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT]:
        player_x -= 5
    if keys[pygame.K_RIGHT]:
        player_x += 5

    # Scherm vullen
    screen.fill(WHITE)

    # Speler tekenen
    pygame.draw.circle(screen, BLUE, (player_x, player_y), 20)

    # Scherm updaten
    pygame.display.flip()

pygame.quit()
sys.exit()

tkinter voor GUI

import tkinter as tk
from tkinter import messagebox

def toon_groet():
    naam = naam_entry.get()
    if naam:
        messagebox.showinfo("Groet", f"Hallo {naam}!")
    else:
        messagebox.showwarning("Waarschuwing", "Voer een naam in!")

# Hoofdvenster maken
root = tk.Tk()
root.title("Eenvoudige GUI")
root.geometry("300x200")

# Widgets toevoegen
naam_label = tk.Label(root, text="Naam:")
naam_label.pack(pady=10)

naam_entry = tk.Entry(root)
naam_entry.pack(pady=5)

groet_button = tk.Button(root, text="Groet!", command=toon_groet)
groet_button.pack(pady=20)

# Applicatie starten
root.mainloop()

Machine Learning Libraries

scikit-learn voor ML

from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score

# Iris dataset laden
iris = load_iris()
X = iris.data  # Features
y = iris.target  # Labels

# Data splitsen in train/test
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Model maken en trainen
knn = KNeighborsClassifier(n_neighbors=3)
knn.fit(X_train, y_train)

# Voorspellingen maken
predictions = knn.predict(X_test)

# Accuracy berekenen
accuracy = accuracy_score(y_test, predictions)
print(f"Model accuracy: {accuracy:.2f}")

TensorFlow/PyTorch voor deep learning

import tensorflow as tf
import numpy as np

# Simpel neuraal netwerk voor XOR probleem
model = tf.keras.Sequential([
    tf.keras.layers.Dense(4, activation='relu', input_shape=(2,)),
    tf.keras.layers.Dense(1, activation='sigmoid')
])

# Model compileren
model.compile(optimizer='adam',
              loss='binary_crossentropy',
              metrics=['accuracy'])

# XOR data
X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([[0], [1], [1], [0]])

# Model trainen
model.fit(X, y, epochs=1000, verbose=0)

# Voorspellingen
predictions = model.predict(X)
print("Voorspellingen voor XOR:")
for i, pred in enumerate(predictions):
    print(f"{X[i]} -> {pred[0]:.3f} (verwacht: {y[i][0]})")
