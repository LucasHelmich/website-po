Wat is localStorage?

localStorage is geheugen van de browser waar je data kunt opslaan. Deze data blijft staan totdat je het verwijdert!

Voordelen:

  • Data blijft bewaard na het sluiten van de browser
  • Elke website heeft zijn eigen aparte opslagruimte
  • Eenvoudig te gebruiken
  • Tot enkele MB per website opslag

Wat je kan opslaan:

  • ‚úÖ Strings (tekst)
  • ‚úÖ JSON (gestructureerde data)
  • ‚úÖ User voorkeur (thema, instellingen)
  • ‚úÖ Formuliergegevens
  • ‚ùå Gevoelige data (wachtwoorden, tokens)

Opslaan & Ophalen

Data persisteren in browser

Data opslaan en ophalen

Data opslaan met setItem()

// Opslaan
localStorage.setItem('naam', 'Lucas');
localStorage.setItem('leeftijd', '16');
localStorage.setItem('stad', 'Amsterdam');

Data ophalen met getItem()

// Ophalen
let naam = localStorage.getItem('naam');
console.log(naam);  // Output: Lucas

let leeftijd = localStorage.getItem('leeftijd');
console.log(leeftijd);  // Output: 16

Controleren of iets bestaat

if (localStorage.getItem('naam') !== null) {
  console.log("Er is een naam opgeslagen!");
} else {
  console.log("Geen naam gevonden");
}

Data verwijderen

// E√©n item verwijderen
localStorage.removeItem('naam');

// Alles verwijderen
localStorage.clear();

JSON - Data structureren

JSON (JavaScript Object Notation) is een formaat om gestructureerde data als tekst op te slaan.

Object naar JSON

// Object
let persoon = {
  naam: "Lucas",
  leeftijd: 16,
  stad: "Amsterdam"
};

// Omzetten naar JSON (string)
let jsonString = JSON.stringify(persoon);
console.log(jsonString);
// Output: {"naam":"Lucas","leeftijd":16,"stad":"Amsterdam"}

JSON naar Object

// JSON string
let jsonString = '{"naam":"Joris","leeftijd":17}';

// Omzetten naar object
let persoon = JSON.parse(jsonString);
console.log(persoon.naam);  // Output: Joris

Arrays naar JSON

// Array
let getallen = [1, 2, 3, 4, 5];

// Als JSON
let json = JSON.stringify(getallen);
console.log(json);  // Output: [1,2,3,4,5]

Complexe data opslaan

Objecten in localStorage opslaan

Het probleem

let gebruiker = { naam: "Lucas", leeftijd: 16 };

// Dit werkt NIET goed:
localStorage.setItem('gebruiker', gebruiker);  // Wordt "[object Object]"!

// Ophalen geeft geen object terug
let opgeslagen = localStorage.getItem('gebruiker');
console.log(opgeslagen);  // "[object Object]" - niet bruikbaar!

De oplossing: JSON stringfy/parse

let gebruiker = { naam: "Lucas", leeftijd: 16 };

// Opslaan als JSON string
localStorage.setItem('gebruiker', JSON.stringify(gebruiker));

// Ophalen en omzetten naar object
let opgeslagen = localStorage.getItem('gebruiker');
let gebruikerObject = JSON.parse(opgeslagen);

console.log(gebruikerObject.naam);  // Lucas
console.log(gebruikerObject.leeftijd);  // 16

Omgaan met lege data

function getUser() {
  let data = localStorage.getItem('gebruiker');
  
  if (data === null) {
    return { naam: "Anoniem", leeftijd: 0 };  // Default waarde
  }
  
  return JSON.parse(data);
}

Praktische voorbeelden

1. Dark mode voorkeur onthouden

// Opslaan
function saveDarkMode(isDark) {
  localStorage.setItem('darkMode', isDark ? 'true' : 'false');
}

// Laden
function loadDarkMode() {
  let isDark = localStorage.getItem('darkMode') === 'true';
  return isDark;
}

2. Todo-lijst opslaan

// Opslaan
function saveTodos(todos) {
  localStorage.setItem('todos', JSON.stringify(todos));
}

// Laden
function loadTodos() {
  let data = localStorage.getItem('todos');
  return data ? JSON.parse(data) : [];
}

// Toevoegen
let todos = loadTodos();
todos.push("Huiswerk maken");
saveTodos(todos);

3. Inloggegevens voorkeur

// Formulier
function onSubmit() {
  let naam = document.getElementById('naam').value;
  let email = document.getElementById('email').value;
  
  let gegevens = { naam, email };
  localStorage.setItem('profile', JSON.stringify(gegevens));
  
  alert('Gegevens opgeslagen!');
}

// Laden bij pagina start
window.addEventListener('load', function() {
  let gegevens = localStorage.getItem('profile');
  if (gegevens) {
    let data = JSON.parse(gegevens);
    document.getElementById('naam').value = data.naam;
    document.getElementById('email').value = data.email;
  }
});

Voorkomen van fouten

  • null - Item bestaat niet, gebruik getItem() result controleren
  • "[object Object]" - Vergeten JSON.stringify() te gebruiken
  • Parse fout - JSON is niet correct, controleer syntax
  • Quota exceeded - Te veel data opgeslagen (zelden)
