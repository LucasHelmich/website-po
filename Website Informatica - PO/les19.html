Bestanden Openen en Sluiten

File handling is essentieel voor het opslaan en laden van data. Python gebruikt het open() statement om bestanden te openen.

Bestand openen - basis

# Bestand openen om te lezen
bestand = open("voorbeeld.txt", "r")  # "r" = read mode

# Bestand openen om te schrijven
bestand = open("nieuw.txt", "w")      # "w" = write mode

# Bestand openen om toe te voegen
bestand = open("log.txt", "a")        # "a" = append mode

# Bestand sluiten
bestand.close()

File modes overzicht

Mode Beschrijving Gebruikt voor
"r" Read (lezen) Bestaande bestanden lezen
"w" Write (schrijven) Nieuwe bestanden maken/overschrijven
"a" Append (toevoegen) Data toevoegen aan bestaande bestanden
"x" Exclusive creation Nieuwe bestanden maken (fout als bestaat)
"r+" Read + Write Lezen en schrijven

Encoding specificeren

# UTF-8 encoding (voor Nederlandse karakters)
bestand = open("tekst.txt", "r", encoding="utf-8")

# Windows-1252 encoding
bestand = open("oud_bestand.txt", "r", encoding="cp1252")

Praktisch voorbeeld

def tel_woorden(bestandsnaam):
    """Telt het aantal woorden in een bestand"""
    try:
        with open(bestandsnaam, "r", encoding="utf-8") as bestand:
            tekst = bestand.read()
            woorden = tekst.split()
            return len(woorden)
    except FileNotFoundError:
        return f"Bestand '{bestandsnaam}' niet gevonden"

print(tel_woorden("voorbeeld.txt"))

Context Managers

Veilig bestanden afhandelen

Context Managers (with statement)

Probleem zonder context manager

# Dit kan problemen veroorzaken!
bestand = open("voorbeeld.txt", "r")
inhoud = bestand.read()
print(inhoud)
# Vergeet bestand te sluiten!
# bestand.close()  # Dit wordt vaak vergeten!

Oplossing met context manager

# Bestand wordt automatisch gesloten
with open("voorbeeld.txt", "r", encoding="utf-8") as bestand:
    inhoud = bestand.read()
    print(inhoud)
# Bestand is hier automatisch gesloten

print("Bestand is nu gesloten")

Voordelen van with statement

  • Automatisch sluiten: Bestand wordt altijd gesloten, zelfs bij errors
  • Minder code: Geen handmatige close() nodig
  • Veiliger: Voorkomt resource leaks
  • Leesbaarder: Duidelijk wanneer bestand gebruikt wordt

Meerdere bestanden tegelijk

with open("input.txt", "r", encoding="utf-8") as input_file, \

    for regel in input_file:
        # Verwerk regel (bijv. maak hoofdletters)
        verwerkte_regel = regel.upper()
        output_file.write(verwerkte_regel)

CSV schrijven

import csv

studenten = [
    ["naam", "leeftijd", "klas"],
    ["Lucas", 16, "4VWO"],
    ["Joris", 17, "5VWO"],
    ["Emma", 16, "4VWO"]
]

with open("nieuwe_studenten.csv", "w", newline="", encoding="utf-8") as csvfile:
    writer = csv.writer(csvfile, delimiter=";")
    for student in studenten:
        writer.writerow(student)

# Met dictionary
studenten_dict = [
    {"naam": "Lucas", "leeftijd": 16, "klas": "4VWO"},
    {"naam": "Joris", "leeftijd": 17, "klas": "5VWO"}
]

with open("studenten_dict.csv", "w", newline="", encoding="utf-8") as csvfile:
    veldnamen = ["naam", "leeftijd", "klas"]
    writer = csv.DictWriter(csvfile, fieldnames=veldnamen, delimiter=";")

    writer.writeheader()
    for student in studenten_dict:
        writer.writerow(student)

JSON Bestanden

JSON importeren

import json

# JSON data
student = {
    "naam": "Lucas",
    "leeftijd": 16,
    "vakken": ["NL", "EN", "WK", "Bio"],
    "cijfers": {
        "NL": 8.5,
        "EN": 7.8,
        "WK": 9.1,
        "Bio": 8.2
    }
}

# Naar JSON string
json_string = json.dumps(student, indent=2)
print(json_string)

# Naar bestand schrijven
with open("student.json", "w", encoding="utf-8") as jsonfile:
    json.dump(student, jsonfile, indent=2)

JSON lezen

import json

# Uit bestand lezen
with open("student.json", "r", encoding="utf-8") as jsonfile:
    geladen_student = json.load(jsonfile)

print(f"Naam: {geladen_student['naam']}")
print(f"Leeftijd: {geladen_student['leeftijd']}")
print(f"Gemiddelde: {sum(geladen_student['cijfers'].values()) / len(geladen_student['cijfers']):.1f}")

# Uit string lezen
json_string = '{"naam": "Joris", "leeftijd": 17}'
data = json.loads(json_string)
print(data)

Complex JSON voorbeeld

import json

# Database van studenten
database = {
    "studenten": [
        {
            "id": 1,
            "naam": "Lucas",
            "vakken": [
                {"naam": "Nederlands", "cijfer": 8.5},
                {"naam": "Engels", "cijfer": 7.8}
            ]
        },
        {
            "id": 2,
            "naam": "Joris",
            "vakken": [
                {"naam": "Nederlands", "cijfer": 7.2},
                {"naam": "Engels", "cijfer": 8.9}
            ]
        }
    ],
    "laatst_bijgewerkt": "2024-01-15"
}

# Opslaan
with open("studenten_database.json", "w", encoding="utf-8") as f:
    json.dump(database, f, indent=2)

# Laden en gemiddelde per student berekenen
with open("studenten_database.json", "r", encoding="utf-8") as f:
    db = json.load(f)

for student in db["studenten"]:
    cijfers = [vak["cijfer"] for vak in student["vakken"]]
    gemiddelde = sum(cijfers) / len(cijfers)
    print(f"{student['naam']}: {gemiddelde:.1f}")
