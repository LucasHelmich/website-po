Wat is Object-Oriented Programming?

Object-Oriented Programming (OOP) is een programmeerparadigma dat draait om objecten. Objecten bundelen data (attributen) en gedrag (methodes) in √©√©n eenheid.

Voordelen van OOP

  • Modulariteit: Code is opgedeeld in herbruikbare objecten
  • Herbruikbaarheid: Klassen kunnen worden uitgebreid en hergebruikt
  • Onderhoudbaarheid: Makkelijker om code te wijzigen en uit te breiden
  • Abstractie: Complexiteit wordt verborgen achter eenvoudige interfaces
  • Real-world modeling: Objecten vertegenwoordigen echte wereld entiteiten

Instance vs Class Variabelen

Instance variabelen

class Student:
    def __init__(self, naam, leeftijd):
        self.naam = naam        # Instance variabele - uniek per object
        self.leeftijd = leeftijd
        self.cijfers = []       # Instance variabele

student1 = Student("Lucas", 16)
student2 = Student("Joris", 17)

student1.cijfers.append(8.5)  # Alleen student1 heeft dit cijfer
student2.cijfers.append(7.2)  # Alleen student2 heeft dit cijfer

print(student1.cijfers)  # [8.5]
print(student2.cijfers)  # [7.2]

Class variabelen

class Student:
    school = "Alberdingk Thijm"  # Class variabele - gedeeld door alle objecten
    aantal_studenten = 0         # Class variabele

    def __init__(self, naam, leeftijd):
        self.naam = naam
        self.leeftijd = leeftijd
        Student.aantal_studenten += 1  # Verhogen bij elke nieuwe student

# Alle objecten delen dezelfde class variabelen
student1 = Student("Lucas", 16)
student2 = Student("Joris", 17)

print(student1.school)           # Alberdingk Thijm
print(student2.school)           # Alberdingk Thijm
print(Student.aantal_studenten)  # 2

# Class variabele wijzigen
Student.school = "Nieuwe School"
print(student1.school)  # Nieuwe School (veranderd voor alle objecten)

Instance methodes vs Class methodes

class Student:
    school = "Alberdingk Thijm"
    aantal_studenten = 0

    def __init__(self, naam):
        self.naam = naam
        Student.aantal_studenten += 1

    def info(self):  # Instance methode
        return f"Naam: {self.naam}, School: {self.school}"

    @classmethod
    def totaal_aantal(cls):  # Class methode
        return f"Totaal aantal studenten: {cls.aantal_studenten}"

    @staticmethod
    def school_info():  # Static methode
        return "Dit is Alberdingk Thijm school"

student1 = Student("Lucas")
print(student1.info())           # Instance methode
print(Student.totaal_aantal())   # Class methode
print(Student.school_info())     # Static methode

Overerving (Inheritance)

Basis overerving

class Dier:
    def __init__(self, naam):
        self.naam = naam

    def maak_geluid(self):
        pass  # Abstracte methode

    def info(self):
        return f"Ik ben {self.naam}"

class Hond(Dier):  # Hond erft van Dier
    def maak_geluid(self):
        return "Woef!"

class Kat(Dier):   # Kat erft van Dier
    def maak_geluid(self):
        return "Miauw!"

# Gebruiken
hond = Hond("Buddy")
kat = Kat("Whiskers")

print(hond.info())        # Ik ben Buddy
print(hond.maak_geluid()) # Woef!
print(kat.info())         # Ik ben Whiskers
print(kat.maak_geluid())  # Miauw!

Overerving met extra functionaliteit

class Voertuig:
    def __init__(self, merk, model):
        self.merk = merk
        self.model = model
        self.snelheid = 0

    def starten(self):
        return f"{self.merk} {self.model} start op"

    def stoppen(self):
        self.snelheid = 0
        return f"{self.merk} {self.model} stopt"

class Auto(Voertuig):
    def __init__(self, merk, model, aantal_deuren):
        super().__init__(merk, model)  # Roep parent constructor aan
        self.aantal_deuren = aantal_deuren
        self.brandstof = 100

    def rijden(self, afstand):
        if self.brandstof > 0:
            self.snelheid = 60
            self.brandstof -= afstand * 0.1
            return f"Rijdt {afstand}km, resterende brandstof: {self.brandstof:.1f}%"
        return "Geen brandstof meer!"

auto = Auto("Toyota", "Corolla", 4)
print(auto.starten())        # Toyota Corolla start op
print(auto.rijden(50))       # Rijdt 50km, resterende brandstof: 95.0%
print(auto.stoppen())        # Toyota Corolla stopt

Multiple inheritance

class Vliegend:
    def vlieg(self):
        return "Ik kan vliegen!"

class Zwemmend:
    def zwem(self):
        return "Ik kan zwemmen!"

class Eend(Vliegend, Zwemmend):  # Multiple inheritance

        self.naam = naam

eend = Eend("Donald")
print(eend.vlieg())  # Ik kan vliegen!
print(eend.zwem())   # Ik kan zwemmen!

Polymorfisme

Method overriding

class Dier:
    def maak_geluid(self):
        return "Geluid..."

class Hond(Dier):
    def maak_geluid(self):  # Override parent methode
        return "Woef!"

class Kat(Dier):
    def maak_geluid(self):  # Override parent methode
        return "Miauw!"

# Polymorfisme - dezelfde methode, verschillend gedrag
dieren = [Hond("Buddy"), Kat("Whiskers"), Dier("Onbekend")]

for dier in dieren:
    print(f"{dier.__class__.__name__}: {dier.maak_geluid()}")

# Output:
# Hond: Woef!
# Kat: Miauw!
# Dier: Geluid...

Duck typing

class Auto:
    def beweeg(self):
        return "Rijdt op de weg"

class Boot:
    def beweeg(self):
        return "Vaart op het water"

class Vliegtuig:
    def beweeg(self):
        return "Vliegt door de lucht"

def start_rit(voertuig):
    """Functie werkt met elk object dat een beweeg() methode heeft"""
    return voertuig.beweeg()

# Polymorfisme door duck typing
voertuigen = [Auto(), Boot(), Vliegtuig()]

for voertuig in voertuigen:
    print(start_rit(voertuig))

# Output:
# Rijdt op de weg
# Vaart op het water
# Vliegt door de lucht

Private attributen

class BankAccount:
    def __init__(self, eigenaar, saldo=0):
        self.eigenaar = eigenaar
        self.__saldo = saldo  # Private attribute (naam begint met __)

    def stort(self, bedrag):
        if bedrag > 0:
            self.__saldo += bedrag
            print(f"‚Ç¨{bedrag} gestort")

    def neem_op(self, bedrag):
        if 0 < bedrag <= self.__saldo:
            self.__saldo -= bedrag
            print(f"‚Ç¨{bedrag} opgenomen")
        else:
            print("Onvoldoende saldo of ongeldig bedrag")

    def get_saldo(self):  # Getter methode
        return self.__saldo

account = BankAccount("Lucas", 1000)
account.stort(500)           # ‚Ç¨500 gestort
account.neem_op(200)         # ‚Ç¨200 opgenomen
print(account.get_saldo())   # 1300

# Direct toegang tot private attribute (werkt niet goed)

print(account._BankAccount__saldo)  # 1300 (name mangling)

Properties

class Student:
    def __init__(self, naam):
        self._naam = naam  # Protected attribute (√©√©n underscore)

    @property
    def naam(self):  # Getter
        return self._naam

    @naam.setter
    def naam(self, waarde):  # Setter
        if isinstance(waarde, str) and len(waarde) > 0:
            self._naam = waarde
        else:
            raise ValueError("Naam moet een niet-lege string zijn")

student = Student("Lucas")
print(student.naam)     # Lucas (via getter)

student.naam = "Joris"  # Via setter

Speciale methodes

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):  # Voor print() en str()
        return f"Vector({self.x}, {self.y})"

    def __repr__(self):  # Voor debugging
        return f"Vector(x={self.x}, y={self.y})"

    def __add__(self, other):  # Voor + operator
        return Vector(self.x + other.x, self.y + other.y)

    def __eq__(self, other):  # Voor == operator
        return self.x == other.x and self.y == other.y

v1 = Vector(1, 2)
v2 = Vector(3, 4)

print(v1)           # Vector(1, 2)
print(repr(v1))     # Vector(x=1, y=2)
print(v1 + v2)      # Vector(4, 6)
print(v1 == v2)     # False
