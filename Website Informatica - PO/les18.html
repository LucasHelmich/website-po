Key-Value Stores

Dictionaries en Sets

Wat zijn Dictionaries?

Dictionaries zijn veranderlijke datastructuren die key-value paren bevatten. Ze worden gebruikt om data te organiseren waar je snel kunt opzoeken op basis van een sleutel.

Dictionaries maken

# Lege dictionary
lege_dict = {}

# Dictionary met data
student = {
    "naam": "Lucas",
    "leeftijd": 16,
    "klas": "4VWO",
    "gemiddelde": 8.5
}

# Dictionary constructor
dict_constructor = dict(naam="Joris", leeftijd=17, stad="Amsterdam")

# Dictionary van lists
telefoonboek = {
    "Lucas": "06-12345678",
    "Joris": "06-87654321",
    "Emma": "06-11223344"
}

print(lege_dict)
print(student)
print(dict_constructor)
print(telefoonboek)

Toegang tot waarden

student = {
    "naam": "Lucas",
    "leeftijd": 16,
    "klas": "4VWO",
    "gemiddelde": 8.5
}

# Toegang via key
print(student["naam"])      # Lucas
print(student["leeftijd"])  # 16

# get() methode (veiliger)
print(student.get("klas"))      # 4VWO
print(student.get("adres", "Onbekend"))  # Onbekend (default waarde)

# Keys, values en items
print(student.keys())    # dict_keys(['naam', 'leeftijd', 'klas', 'gemiddelde'])
print(student.values())  # dict_values(['Lucas', 16, '4VWO', 8.5])
print(student.items())   # dict_items([('naam', 'Lucas'), ('leeftijd', 16), ...])

Dictionary Operaties

Bewerken van dictionaries

Dictionary Operaties

Waarden toevoegen/wijzigen

student = {"naam": "Lucas", "leeftijd": 16}

# Nieuwe key-value paar toevoegen
student["klas"] = "4VWO"
print(student)  # {'naam': 'Lucas', 'leeftijd': 16, 'klas': '4VWO'}

# Bestaande waarde wijzigen
student["leeftijd"] = 17
print(student)  # {'naam': 'Lucas', 'leeftijd': 17, 'klas': '4VWO'}

# update() methode
student.update({"gemiddelde": 8.5, "vakken": ["NL", "EN", "WK"]})
print(student)

Waarden verwijderen

student = {
    "naam": "Lucas",
    "leeftijd": 17,
    "klas": "4VWO",
    "gemiddelde": 8.5
}

# pop() - verwijdert en retourneert waarde
leeftijd = student.pop("leeftijd")
print(f"Verwijderde leeftijd: {leeftijd}")  # 17
print(student)

# popitem() - verwijdert laatste key-value paar
laatste = student.popitem()
print(f"Laatste item: {laatste}")  # ('gemiddelde', 8.5)

# del statement
del student["klas"]
print(student)  # {'naam': 'Lucas'}

# clear() - maakt dictionary leeg
student.clear()
print(student)  # {}

Controleren of key bestaat

student = {"naam": "Lucas", "leeftijd": 16}

# in operator
print("naam" in student)        # True
print("adres" in student)       # False

# not in operator
print("adres" not in student)   # True

# Veilige toegang
if "leeftijd" in student:
    print(f"Leeftijd: {student['leeftijd']}")
else:
    print("Leeftijd niet bekend")

# Of met get()
leeftijd = student.get("leeftijd", "Onbekend")
print(f"Leeftijd: {leeftijd}")

Dictionary Iteratie

Itereren over dictionaries

student = {
    "naam": "Lucas",
    "leeftijd": 16,
    "klas": "4VWO",
    "vakken": ["NL", "EN", "WK"]
}

# Itereren over keys
print("Keys:")
for key in student:
    print(f"  {key}")

# Itereren over keys (expliciet)
print("\nKeys (expliciet):")
for key in student.keys():
    print(f"  {key}")

# Itereren over values
print("\nValues:")
for value in student.values():
    print(f"  {value}")

# Itereren over items (key-value paren)
print("\nKey-Value paren:")
for key, value in student.items():
    print(f"  {key}: {value}")

Praktische voorbeelden

# Studenten database
studenten = {
    "Lucas": {"leeftijd": 16, "gemiddelde": 8.5},
    "Joris": {"leeftijd": 17, "gemiddelde": 7.8},
    "Emma": {"leeftijd": 16, "gemiddelde": 9.2}
}

# Gemiddelde leeftijd berekenen
totaal_leeftijd = 0
for student_data in studenten.values():
    totaal_leeftijd += student_data["leeftijd"]

gemiddelde_leeftijd = totaal_leeftijd / len(studenten)
print(f"Gemiddelde leeftijd: {gemiddelde_leeftijd}")

# Studenten met gemiddelde >= 8.0
print("\nGoede studenten:")
for naam, data in studenten.items():
    if data["gemiddelde"] >= 8.0:
        print(f"  {naam}: {data['gemiddelde']}")

Unieke verzamelingen

Wat zijn Sets?

Sets maken

# Lege set
lege_set = set()

# Set met waarden
kleuren = {"rood", "groen", "blauw", "rood"}  # Duplicaat wordt verwijderd
print(kleuren)  # {'blauw', 'groen', 'rood'} (volgorde niet gegarandeerd)

# Set van list
getallen = set([1, 2, 3, 2, 1])  # Duplicaten worden verwijderd
print(getallen)  # {1, 2, 3}

# Set van string
letters = set("hallo")  # Elke letter √©√©n keer
print(letters)  # {'l', 'o', 'h', 'a'}

Set operaties

set_a = {1, 2, 3, 4}
set_b = {3, 4, 5, 6}

# Union (vereniging) - alle elementen uit beide sets
print("Union:", set_a | set_b)        # {1, 2, 3, 4, 5, 6}
print("Union:", set_a.union(set_b))   # {1, 2, 3, 4, 5, 6}

# Intersection (doorsnede) - elementen die in beide sets zitten
print("Intersection:", set_a & set_b)           # {3, 4}
print("Intersection:", set_a.intersection(set_b))  # {3, 4}

# Difference (verschil) - elementen in A maar niet in B
print("Difference A-B:", set_a - set_b)         # {1, 2}
print("Difference B-A:", set_b - set_a)         # {5, 6}

# Symmetric difference - elementen in A of B maar niet in beide
print("Symmetric diff:", set_a ^ set_b)         # {1, 2, 5, 6}

Set methodes

fruit = {"appel", "banaan", "kers"}

# Element toevoegen
fruit.add("druif")
print(fruit)  # {'kers', 'druif', 'appel', 'banaan'}

# Meerdere elementen toevoegen
fruit.update(["mango", "peer"])
print(fruit)  # {'kers', 'druif', 'peer', 'mango', 'appel', 'banaan'}

# Element verwijderen
fruit.remove("banaan")  # Error als element niet bestaat
print(fruit)

fruit.discard("appel")  # Geen error als element niet bestaat
print(fruit)

# Willekeurig element verwijderen
willekeurig = fruit.pop()
print(f"Verwijderd: {willekeurig}")

# Set leegmaken
fruit.clear()
print(fruit)  # set()

Dictionary Comprehensions

Basis dictionary comprehension

# Traditionele manier
kwadraten = {}
for i in range(1, 6):
    kwadraten[i] = i ** 2
print(kwadraten)  # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

# Met comprehension
kwadraten = {i: i**2 for i in range(1, 6)}
print(kwadraten)  # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

Met voorwaarden

# Alleen even getallen
even_kwadraten = {i: i**2 for i in range(1, 11) if i % 2 == 0}
print(even_kwadraten)  # {2: 4, 4: 16, 6: 36, 8: 64, 10: 100}

# Omgekeerde mapping
namen = ["Lucas", "Joris", "Emma"]
naam_lengtes = {naam: len(naam) for naam in namen}
print(naam_lengtes)  # {'Lucas': 5, 'Joris': 5, 'Emma': 4}

Uit bestaande dictionary

studenten = {
    "Lucas": 8.5,
    "Joris": 7.2,
    "Emma": 9.1,
    "Anna": 6.8
}

# Alleen geslaagde studenten
geslaagd = {naam: cijfer for naam, cijfer in studenten.items() if cijfer >= 7.0}
print(geslaagd)  # {'Lucas': 8.5, 'Joris': 7.2, 'Emma': 9.1}

# Cijfers naar letters
def cijfer_naar_letter(cijfer):
    if cijfer >= 9.0: return "A"
    elif cijfer >= 8.0: return "B"
    elif cijfer >= 7.0: return "C"
    else: return "D"

letters = {naam: cijfer_naar_letter(cijfer) for naam, cijfer in studenten.items()}
print(letters)  # {'Lucas': 'B', 'Joris': 'C', 'Emma': 'A', 'Anna': 'D'}

Set Comprehensions

Basis set comprehension

# Traditionele manier
kwadraten = set()
for i in range(1, 6):
    kwadraten.add(i ** 2)
print(kwadraten)  # {1, 4, 9, 16, 25}

# Met comprehension
kwadraten = {i**2 for i in range(1, 6)}
print(kwadraten)  # {1, 4, 9, 16, 25}

Met voorwaarden

# Even kwadraten
even_kwadraten = {i**2 for i in range(1, 11) if i % 2 == 0}
print(even_kwadraten)  # {4, 16, 36, 64, 100}

# Unieke eerste letters
namen = ["Lucas", "Joris", "Emma", "Anna", "Lisa"]
eerste_letters = {naam[0] for naam in namen}
print(eerste_letters)  # {'L', 'J', 'E', 'A'}

Praktische toepassingen

# Unieke woorden uit tekst
tekst = "het huis is groot en het huis is mooi"
woorden = tekst.split()
unieke_woorden = {woord.lower() for woord in woorden}
print(unieke_woorden)  # {'het', 'huis', 'is', 'groot', 'en', 'mooi'}

# Getallen die deelbaar zijn door 2, 3 of 5
getallen = {i for i in range(1, 21) if i % 2 == 0 or i % 3 == 0 or i % 5 == 0}
print(getallen)  # {2, 3, 4, 5, 6, 8, 9, 10, 12, 14, 15, 16, 18, 20}

# Lengtes van woorden
woorden = ["Python", "is", "een", "geweldige", "programmeertaal"]
woord_lengtes = {len(woord) for woord in woorden}
print(woord_lengtes)  # {2, 3, 6, 9, 18}
